Eventos - event.type

JavaScript

<script type="text/javascript">
window.onload = function() {	
	var tipoEvento = function(event){
		switch(event.type) {
			case "click":
				tudo.style.background = "red";
				break;
			case "dblclick":
				tudo.style.background = "";
				break;
			case "mouseover":
				this.style.background = "blue";
				break;
			case "mouseout":
				this.style.background = "";
				break;
		}
	};	
	var tudo = document.getElementById("tudo");	
	var botao = document.getElementById("demo");	
	botao.onclick = tipoEvento;
	botao.onmouseover = tipoEvento;
	botao.onmouseout = tipoEvento;
	botao.ondblclick = tipoEvento;
}
</script>
