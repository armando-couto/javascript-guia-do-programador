Propriedades do objeto meta

JavaScript

<script type="text/javascript">
window.onload = function () {
var el, msg, mensagem, elBotao;
	el = document.getElementsByTagName("meta")
	msg = document.getElementById("msg")
	elBotao = document.getElementById("btn");	
	mensagem = "<h3>Atributos do primeiro elemento <code>meta</code> do documento</h3>";
	mensagem += "<ul>";
	mensagem += "<li>http-equiv= " + el[0].httpEquiv + "</li>";
	mensagem += "<li>content= " + el[0].content + "</li>";
	mensagem += "</ul>";	
	elBotao.onclick	= function() {
		msg.innerHTML = mensagem;
	}
}
<script>
