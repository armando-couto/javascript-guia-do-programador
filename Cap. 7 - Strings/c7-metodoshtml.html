JavaScript

<script type="text/javascript">
	var s = 'javascript';
	function metodoHTML(metodo, arg) {
		var marcacao = document.getElementById('marcacao');
		
		switch (metodo) {
			case 'ancora':
			document.form.resultado.value = s.anchor(arg);
			marcacao.innerHTML = s.anchor(arg);
			break;
		
			case 'big':
			document.form.resultado.value = s.big();
			marcacao.innerHTML = s.big();
			break;
		
			case 'bold':
			document.form.resultado.value = s.bold();
			marcacao.innerHTML = s.bold();
			break;
		
			case 'fixed':
			document.form.resultado.value = s.fixed();
			marcacao.innerHTML = s.fixed();
			break;
		
			case 'fontcolor':
			document.form.resultado.value = s.fontcolor(arg);
			marcacao.innerHTML = s.fontcolor(arg);
			break;
			
			case 'fontsize':
			document.form.resultado.value = s.fontsize(7);
			marcacao.innerHTML = s.fontsize(7);
			break;

			case 'italics':
			document.form.resultado.value = s.italics();
			marcacao.innerHTML = s.italics();
			break;

			case 'link':
			document.form.resultado.value = s.link(arg);
			marcacao.innerHTML = s.link(arg);
			break;

			case 'small':
			document.form.resultado.value = s.small();
			marcacao.innerHTML = s.small();
			break;

			case 'strike':
			document.form.resultado.value = s.strike();
			marcacao.innerHTML = s.strike();
			break;

			case 'sub':
			document.form.resultado.value = s.sub();
			marcacao.innerHTML = s.sub();
			break;

			case 'sup':
			document.form.resultado.value = s.sup();
			marcacao.innerHTML = s.sup();
			break;
		};
	}<script>
